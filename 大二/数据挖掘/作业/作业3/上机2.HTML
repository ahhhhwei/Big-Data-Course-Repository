def get_accuracy(y:list, y_predicted:list):
    sum = 0
    for i in zip(y, y_predicted):
        if i[0] == i[1]:
            sum += 1
    sum /= len(y)
    return sum

def get_precision(y:list, y_predicted:list):
    TP = 0  # 真正例
    TN = 0  # 真反例
    FP = 0  # 假正例
    FN = 0  # 假反例
    for i in zip(y, y_predicted):
        if i[0] == i[1] == 1:
            TP += 1
        elif i[0] == i[1] == 0:
            TN += 1
        elif i[0] == 0 and i[1] == 1:
            FP += 1
        elif i[0] == 1 and i[1] == 0:
            FN += 1
    return TP / (TP + FP)

def get_F1_measure(y:list, y_predicted:list):
    P = get_precision(y, y_predicted)
    R = get_recall(y, y_predicted)
    F1 = 2 * P * R / (P + R)
    return F1

import random
# 留出法
def StrateSplit(x, y:list, test_ratio:float, random_state:float):
    X = x.copy(deep=True)
    random.seed(random_state)
    index = len(X.T)
    X.insert(index, 'target', y)
    temp = X.values
    random.shuffle(temp)
    lenth = int(test_ratio * len(temp))
    y_train = [row[index:] for row in temp[lenth:]]
    y_test = [row[index:] for row in temp[:lenth]]
    X_train = [row[:index] for row in temp[lenth:]]
    X_test = [row[:index] for row in temp[:lenth]]
    X_test = pd.DataFrame(X_test)
    y_test = pd.DataFrame(y_test)
    X_train = pd.DataFrame(X_train)
    y_train = pd.DataFrame(y_train)
    return X_train, y_train, X_test, y_test

# 自助法
def BoostTr(x, y, random_state:int):
    X = x.copy(deep=True)
    X_test = []
    n = len(X)
    random.seed(random_state)
    index = len(X.T)
    temp = X.values
    X_train = X.sample(n, replace=True)
    for i in list(X.values):
        flag = True
        for j in list(X_train.values):
            if (i == j).all():
                flag = False
                break
        if flag:
            X_test.append(i)
    X_test = pd.DataFrame(X_test)
    y_train = X_train.iloc[:,4:]
    X_train = X_train.iloc[:,:4]
    y_test = X_test.iloc[:,4:]
    X_test = X_test.iloc[:,:4]
    return X_train, y_train, X_test, y_test

import numpy as np
from sklearn.model_selection import KFold

def KfoldSplit(x, k, random_state):
    n_sample = x.copy(deep=True)
    kf = KFold(n_splits=k, shuffle=True, random_state=random_state)
    folds_index = []
    for test_index in kf.split(n_sample):
         folds_index.append(test_index)
    return pd.DataFrame(folds_index)

3、 编程实现 k 最近邻分类¶

  • 编写 k-nearestN 函数，输入训练数据 X_train,其对应标签 y,近邻数 k，测试集 X_test，以及距离度量，输出测试集中每个样本的标签。默认距离度量为欧式距离，其他可选距离包括余弦相似度等。

# 编造一些数据
train = [(2, 0), (2, 1), (0, 3), (1, 5), (1, 4), (2, 2), (2, 3), (2, 4)]
labels = ['red', 'red', 'blue', 'blue', 'red', 'blue', 'red', 'blue']
test = [(1, 1), (3, 3), (1, 3)]
def distance(x1, y1, x2, y2, distancetype):
    # 计算欧氏距离
    if distancetype == 'Euclidean':
        return ((x1 - x2)**2 + (y1 - y2)**2)**0.5
    # 计算余弦相似度
    elif distancetype == 'Cosline':
        return (x1*y1 + x2*y2)/((x1**2+y1**2)**(0.5+x2**2+y2**2)**0.5)
    # 计算曼哈顿距离
    elif distancetype == 'Manhattan':
        return (x1-y1) + (x2-y2)
    elif distancetype == 'Chebyshev':
        return max(x1-y1, x2-y2)
    # 其他各种距离同理
    
def k_nearestN(X_train, y, k, X_test:list, distancetype = 'Euclidean'):
    ret_labels = []
    for i in X_test:
        hash_table = {}
        temp = []
        for j in X_train:
            # 距离添加进列表
            temp.append(distance(i[0], i[1], j[0], j[1], distancetype))
        # 打包训练集的点和对应的距离
        temp = list(zip(temp, X_train, y))
        # 按每个元组的第一个元素(距离)排升序
        temp.sort(key = lambda x:x[0])
        temp = temp[0 : k]
        for e in temp:
            if e[2] in hash_table:
                hash_table[e[2]] += 1
            else:
                hash_table[e[2]] = 1
        max_value = 0
        max_key = ''
        for e in hash_table:
            if hash_table[e] > max_value:
                max_value = hash_table[e]
                max_key = e
        ret_labels.append(max_key)
    return ret_labels

import numpy as np
import pandas as pd
from sklearn.model_selection import KFold

def KfoldSplit(x, k, random_state, y):
    n_sample = x.copy(deep=True)
    kf = KFold(n_splits=k, shuffle=True, random_state=random_state)
    sample_index = []
    label_index = []
    for X_index, y_index in kf.split(n_sample, y):
        sample_index.append(X_index)
        label_index.append(y_index)
    return pd.DataFrame(sample_index).T,pd.DataFrame(label_index).T
